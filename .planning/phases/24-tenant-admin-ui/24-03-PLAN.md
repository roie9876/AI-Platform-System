```
function StepIndicator({ steps, currentStep }: { steps: { label: string; icon: ReactNode }[]; currentStep: number }) {
  return (
    <div className="flex items-center gap-2 mb-8">
      {steps.map((step, i) => (
        <Fragment key={i}>
          {i > 0 && <div className={`h-px flex-1 ${i <= currentStep ? 'bg-blue-600' : 'bg-gray-200'}`} />}
          <div className="flex flex-col items-center gap-1">
            <div className={`flex h-8 w-8 items-center justify-center rounded-full text-sm font-medium ${
              i < currentStep ? 'bg-blue-600 text-white' :
              i === currentStep ? 'border-2 border-blue-600 text-blue-600' :
              'border-2 border-gray-300 text-gray-400'
            }`}>
              {i < currentStep ? <Check className="h-4 w-4" /> : i + 1}
            </div>
            <span className={`text-xs ${i <= currentStep ? 'text-gray-700' : 'text-gray-400'}`}>{step.label}</span>
          </div>
        </Fragment>
      ))}
    </div>
  );
}
```
