```python
from fastapi import APIRouter
from app.repositories.cosmos_client import get_cosmos_client

health_router = APIRouter(tags=["health"])

@health_router.get("/healthz")
async def liveness():
    """Liveness probe — process is alive."""
    return {"status": "ok"}

@health_router.get("/readyz")
async def readiness():
    """Readiness probe — dependencies reachable."""
    try:
        client = await get_cosmos_client()
        # Lightweight check — list databases is cheap
        return {"status": "ok"}
    except Exception:
        return JSONResponse(status_code=503, content={"status": "unavailable"})

@health_router.get("/startupz")
async def startup():
    """Startup probe — application initialized."""
    return {"status": "ok"}
```

**2. Create `backend/app/services/service_client.py`** — Inter-service HTTP client:

```python
import httpx
import os
from typing import Any, Dict, Optional

class ServiceClient:
    """HTTP client for inter-service communication within AKS cluster."""
    
    TOOL_EXECUTOR_URL = os.environ.get("TOOL_EXECUTOR_URL", "http://tool-executor:8000")
    MCP_PROXY_URL = os.environ.get("MCP_PROXY_URL", "http://mcp-proxy:8000")
    AGENT_EXECUTOR_URL = os.environ.get("AGENT_EXECUTOR_URL", "http://agent-executor:8000")
    API_GATEWAY_URL = os.environ.get("API_GATEWAY_URL", "http://api-gateway:8000")
    WORKFLOW_ENGINE_URL = os.environ.get("WORKFLOW_ENGINE_URL", "http://workflow-engine:8000")

    def __init__(self):
        self._client = httpx.AsyncClient(timeout=120.0)

    async def execute_tool(self, tool_name: str, input_data: dict, input_schema: dict,
                           execution_command: Optional[str], timeout_seconds: int,
                           auth_token: str) -> dict:
        """Call tool-executor to run a tool."""
        resp = await self._client.post(
            f"{self.TOOL_EXECUTOR_URL}/api/v1/internal/tools/execute",
            json={"tool_name": tool_name, "input_data": input_data,
                  "input_schema": input_schema, "execution_command": execution_command,
                  "timeout_seconds": timeout_seconds},
            headers={"Authorization": f"Bearer {auth_token}"},
        )
        resp.raise_for_status()
        return resp.json()

    async def call_mcp_tool(self, server_url: str, tool_name: str, arguments: dict,
                            auth_headers: Optional[dict], auth_token: str) -> dict:
        """Call mcp-proxy to invoke an MCP tool."""
        resp = await self._client.post(
            f"{self.MCP_PROXY_URL}/api/v1/internal/mcp/call-tool",
            json={"server_url": server_url, "tool_name": tool_name,
                  "arguments": arguments, "auth_headers": auth_headers},
            headers={"Authorization": f"Bearer {auth_token}"},
        )
        resp.raise_for_status()
        return resp.json()

    async def execute_agent(self, agent_id: str, message: str, tenant_id: str,
                            user_id: str, thread_id: Optional[str],
                            auth_token: str) -> dict:
        """Call agent-executor to run an agent (used by workflow-engine)."""
        resp = await self._client.post(
            f"{self.AGENT_EXECUTOR_URL}/api/v1/internal/agents/{agent_id}/execute",
            json={"message": message, "tenant_id": tenant_id,
                  "user_id": user_id, "thread_id": thread_id},
            headers={"Authorization": f"Bearer {auth_token}"},
        )
        resp.raise_for_status()
        return resp.json()

    async def close(self):
        await self._client.aclose()
```
