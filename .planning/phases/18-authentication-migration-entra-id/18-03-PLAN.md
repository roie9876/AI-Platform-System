    1. Add to `backend/app/core/security.py`:
       ```python
       from azure.identity.aio import DefaultAzureCredential

       _credential: DefaultAzureCredential | None = None

       def get_azure_credential() -> DefaultAzureCredential:
           """Get or create a DefaultAzureCredential instance.
           Works via Azure CLI locally and Workload Identity on AKS.
           """
           global _credential
           if _credential is None:
               _credential = DefaultAzureCredential()
           return _credential

       async def get_service_token(scope: str) -> str:
           """Acquire a token for a downstream Azure service.
           Args:
               scope: The resource scope, e.g. "https://cosmos.azure.com/.default"
           Returns:
               Access token string
           """
           credential = get_azure_credential()
           token = await credential.get_token(scope)
           return token.token
       ```
